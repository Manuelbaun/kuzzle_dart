Implementation 

Future<User> createUser(
  User user,
  Credentials credentials, {
  String refresh = 'false',
}) =>
    addNetworkQuery(
      'createUser',
      body: <String, dynamic>{
        'content': user.toMap(),
        'credentials': credentials.toMap(),
      },
      optionalParams: <String, dynamic>{
        'refresh': refresh,
      },
    ).then((response) => User.fromMap(this, response.result));
