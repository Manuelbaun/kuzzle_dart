Map<String, dynamic> getPartialQuery() {
  final query = <String, dynamic>{
    'controller': getController(),
  };
  if (collection != null) {
    query.addAll(<String, dynamic>{
      'index': collection.index,
      'collection': collection.collectionName,
    });
  } else if (kuzzle != null) {
    query.addAll(<String, dynamic>{
      'index': kuzzle.defaultIndex,
    });
  }
  return query;
}
